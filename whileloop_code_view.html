<h3>Loops: while : Block 1</h3>
	<?php
		$count = 0;

		while ($count <= 10) {
			echo $count . " , ";
			$count++;
		}
		echo "Total Count: {$count}";
	?>
	<h3>Loops: while: Block 2</h3>
	<?php
		$count = 0;

		while ($count <= 10) {

			if ($count == 5) {
				echo " Five ";
			}
			else {
				echo $count . " , ";
			}
			$count++;
		}
		echo "Total Count: {$count}";
	?>
