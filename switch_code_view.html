<h3>switch: Block</h3>
<?php
$a = 3;

switch ($a) {
	case 0:
		echo "a is equal to 0";
		break;
	case 1:
		echo "a is equal to 1";
		break;
	case 2:
		echo "a is equal to 2";
		break;
	default:
		echo "a is not equal to 0, 1 or 2";
		break;
}

?>
<!-- PHP SECTION switch END -->
